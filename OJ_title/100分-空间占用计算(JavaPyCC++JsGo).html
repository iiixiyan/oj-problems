100分-空间占用计算

员工A的磁盘空间经常被耗尽，他需要找到占用空间最大的目录或文件，然后决定如何清理文件释放空间。
给定某一目录，请编写程序帮助他统计该目录内一级子目录和文件的占用空间，并返回目标目录一级子项（文件或子目录）中占用空间最大的项。

规则说明：

  1. 目录占用空间为其内部所有文件 Size 的总和，目录本身 Size 为 0。
  2. 目录深度不高于 7 层，目录或文件名总长度不超过 128 字节。
  3. 当存在多个子项占用空间均为最大时，多个子项采用字符升序排列。
  4. 目标目录不在文件系统中时（输入路径前缀匹配不到任何路径），返回空列表。

输入：
参数1：要进行统计的目标目录。
参数2：文件系统内的文件列表。
参数3：文件 Size 列表，该列表中的数据和文件列表存在一一对应关系。

示例1

输入
"/dir1/dir2-1", ["/dir0/dir1-1/file1-1", "/dir1/dir1-1/file1-1", "/dir1/dir2-1/file3-1", "/dir1/dir2-1/file3-2", "/dir1/dir2-1/dir3-1/file4-1"], [8192, 81920, 2048, 8192, 1024]

输出
["/dir1/dir2-1/file3-2"]

说明
/dir1/dir2-1 下共有三个一级子项 file3-1、file3-2、dir3-1，它们的占用空间统计如下：

file3-1 : 2048
file3-2 : 8192
dir3-1 : 1024，其内部所有文件的 size 之和，由于内部只有一个文件 file4-1，因此和 file4-1 的 size 一致。

其中 file3-2 占用的空间最大，因此返回 ["/dir1/dir2-1/file3-2"]

示例2

输入
"/dir1", ["/dir0/dir2-1/file3-1", "/dir1/dir2-1/file3-1", "/dir1/dir2-1/file3-2", "/dir1/dir2-2/file3-3", "/dir1/file2-3"], [10240, 4096, 8192, 10240, 8192]

输出
["/dir1/dir2-1"]

题解

思路：模拟

  1. 通过输入文件列表，创建出文件树结构。其中文件设置 size 为文件大小，文件夹 size 全部设置为 0
  2. 在创建的文件树中找到 目标目录 子树。
  3. 递归计算子树中子目录的大小。
  4. 合并子目录和子文件，按照 大小降序，名字升序 自定排序。找出所有最大项名称，返回即可。

// 通用 切割函数 函数 将字符串str根据delimiter进行切割

vector<string> split(const string& str, const string& delimiter) {

    vector<string> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(str.substr(start, end - start));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(str.substr(start));

    return result;

}

 

// 通用 切割函数 函数 将字符串str根据delimiter进行切割

vector<int> splitInt(const string& str, const string& delimiter) {

    vector<int> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(stoi(str.substr(start, end - start)));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(stoi(str.substr(start)));

    return result;

}

 

 

 

struct Node {

    map<string, Node*> childrenDir;

    map<string, Node*> childrenFile;

    string name;

    int size;

    Node() {

        size = 0;

    }

};

 

int dfs(Node* root) {

    int sum = 0;

    if (!root->childrenDir.empty()) {

        for (auto &p : root->childrenDir) {

            sum += dfs(p.second);

        }

    }

    // 计算当前文件

    if (!root->childrenFile.empty()) {

        for (auto &p: root->childrenFile) {

            sum += p.second->size;

        }

    }

 

    // 更新当前目录总大小

    root->size = sum;

    return sum;

}

 

bool cmp(Node* &a, Node* &b) {

    if (a->size == b->size) {

        return a->name < b->name;

    }

    return a->size > b->size;

}

 

 

vector<string> total(string& dest, vector<string>& fileList, vector<int>& fileSizeList) {

    vector<string> res;

    Node* root = new Node();

    // 构建目录树

    for (int i = 0; i < fileList.size(); i++) {

        string path = fileList[i];

        int size = fileSizeList[i];

        vector<string> part = split(path, "/");

        Node* tmp = root;

        for (int j = 1; j < part.size(); j++) {

            string name = part[j];

            bool isFile = (j == part.size() -1);

            if (!isFile) {

                if (tmp->childrenDir.find(name) == tmp->childrenDir.end()) {

                    Node* dir = new Node();

                    dir->name = name;

                    tmp->childrenDir[name] = dir;

                }

                tmp = tmp->childrenDir[name];

            } else {

                Node* file = new Node();

                file->name = name;

                file->size= size;

                tmp->childrenFile[name] = file;

            }

        }

    }

   

    // 获取指定目录

    Node* tmp = root;

    vector<string> part = split(dest, "/");

    for (int i = 1; i < part.size(); i++) {

        string name = part[i];

        // 不存在指定目录

        if (tmp->childrenDir.find(name) == tmp->childrenDir.end()) {

            return res;

        } else {

            tmp = tmp->childrenDir[name];

        }

    }

    // 递归计算子目录总大小

    dfs(tmp);

    // 合并指定子目录/文件

    vector<Node*> listFile;

    for (auto &p : tmp->childrenFile) {

        listFile.push_back(p.second);

    }

    for (auto &p : tmp->childrenDir) {

        listFile.push_back(p.second);

    }

    if (listFile.empty()) {

        return res;

    }

    //自定排序

    sort(listFile.begin(), listFile.end(), cmp);

    res.push_back(listFile[0]->name);

    int maxSize = listFile[0]->size;

    for (int i = 1; i < listFile.size(); i++) {

        Node* tmp = listFile[i];

        if (tmp->size != maxSize) {

            break;

        }

        res.push_back(tmp->name);

    }

    return res;

}

import java.util.*;

 

public class Main {

 

    // 目录节点

    static class Node {

        Map<String, Node> childrenDir = new HashMap<>();

        Map<String, Node> childrenFile = new HashMap<>();

        String name;

        int size = 0;

    }

 

    // DFS 计算目录大小

    static int dfs(Node root) {

        int sum = 0;

 

        // 子目录

        for (Node node : root.childrenDir.values()) {

            sum += dfs(node);

        }

 

        // 文件

        for (Node node : root.childrenFile.values()) {

            sum += node.size;

        }

 

        root.size = sum;

        return sum;

    }

 

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

 

        String path = sc.nextLine();

        String file = sc.nextLine();

        String fileSize = sc.nextLine();

 

        String[] fileList = file.split(" ");

        String[] sizeArr = fileSize.split(" ");

 

        Node root = new Node();

 

        // 构建目录树

        for (int i = 0; i < fileList.length; i++) {

            String[] parts = fileList[i].split("/");

            Node cur = root;

 

            for (int j = 1; j < parts.length; j++) {

                String name = parts[j];

                boolean isFile = (j == parts.length - 1);

 

                if (!isFile) {

                    cur.childrenDir.putIfAbsent(name, new Node());

                    cur.childrenDir.get(name).name = name;

                    cur = cur.childrenDir.get(name);

                } else {

                    Node fileNode = new Node();

                    fileNode.name = name;

                    fileNode.size = Integer.parseInt(sizeArr[i]);

                    cur.childrenFile.put(name, fileNode);

                }

            }

        }

 

        // 找目标目录

        Node cur = root;

        String[] parts = path.split("/");

 

        for (int i = 1; i < parts.length; i++) {

            if (!cur.childrenDir.containsKey(parts[i])) {

                return;

            }

            cur = cur.childrenDir.get(parts[i]);

        }

 

        dfs(cur);

 

        List<Node> list = new ArrayList<>();

        list.addAll(cur.childrenDir.values());

        list.addAll(cur.childrenFile.values());

 

        if (list.isEmpty()) return;

 

        // 排序

        list.sort((a, b) -> {

            if (a.size == b.size) return a.name.compareTo(b.name);

            return b.size - a.size;

        });

 

        int max = list.get(0).size;

 

        for (Node node : list) {

            if (node.size != max) break;

            System.out.print(node.name + " ");

        }

    }

}

package main

 

import (

       "bufio"

       "fmt"

       "os"

type Node struct {

       childrenDir  map[string]*Node

       childrenFile map[string]*Node

       name         string

       size         int

}

 

// DFS计算

func dfs(root *Node) int {

       sum := 0

 

       for _, node := range root.childrenDir {

              sum += dfs(node)

       }

 

       for _, node := range root.childrenFile {

              sum += node.size

       }

 

       root.size = sum

       return sum

}

 

func main() {

       scanner := bufio.NewScanner(os.Stdin)

 

       scanner.Scan()

       path := scanner.Text()

 

       scanner.Scan()

       fileList := strings.Split(scanner.Text(), " ")

 

       scanner.Scan()

       sizeStr := strings.Split(scanner.Text(), " ")

 

       root := &Node{

              childrenDir:  make(map[string]*Node),

              childrenFile: make(map[string]*Node),

       }

 

       // 构建目录树

       for i := 0; i < len(fileList); i++ {

              parts := strings.Split(fileList[i], "/")

              cur := root

 

              for j := 1; j < len(parts); j++ {

                     name := parts[j]

                     isFile := (j == len(parts)-1)

 

                     if !isFile {

                            if _, ok := cur.childrenDir[name]; !ok {

                                   cur.childrenDir[name] = &Node{

                                          childrenDir:  make(map[string]*Node),

                                          childrenFile: make(map[string]*Node),

                                          name:         name,

                                   }

                            }

                            cur = cur.childrenDir[name]

                     } else {

                            size := 0

                            fmt.Sscanf(sizeStr[i], "%d", &size)

 

                            cur.childrenFile[name] = &Node{

                                   name: name,

                                   size: size,

                            }

                     }

              }

       }

 

       // 找目录

       cur := root

       for _, name := range strings.Split(path, "/")[1:] {

              if _, ok := cur.childrenDir[name]; !ok {

                     return

              }

              cur = cur.childrenDir[name]

       }

 

       dfs(cur)

 

       list := []*Node{}

       for _, v := range cur.childrenDir {

              list = append(list, v)

       }

       for _, v := range cur.childrenFile {

              list = append(list, v)

       }

 

       if len(list) == 0 {

              return

       }

 

       // 排序

       for i := 0; i < len(list); i++ {

              for j := i + 1; j < len(list); j++ {

                     if list[i].size < list[j].size ||

                            (list[i].size == list[j].size && list[i].name > list[j].name) {

                            list[i], list[j] = list[j], list[i]

                     }

              }

       }

 

       max := list[0].size

 

       for _, node := range list {

              if node.size != max {

                     break

              }

              fmt.Print(node.name + " ")

       }

#include <stdio.h>

#include <stdlib.h>

#include <string.h>

 

#define MAX_CHILD 100   // 每个目录最多子节点

#define MAX_NAME  129

 

// 定义节点

typedef struct Node {

    char name[MAX_NAME];

    int size;

 

    struct Node* childrenDir[MAX_CHILD];   // 子目录

    struct Node* childrenFile[MAX_CHILD];  // 文件

 

    int dirCount;

    int fileCount;

} Node;

 

// 创建节点

Node* createNode(char* name) {

    Node* node = (Node*)malloc(sizeof(Node));

    strcpy(node->name, name);

    node->size = 0;

    node->dirCount = 0;

    node->fileCount = 0;

    return node;

}

 

// 查找子目录

Node* findDir(Node* cur, char* name) {

    for (int i = 0; i < cur->dirCount; i++) {

        if (strcmp(cur->childrenDir[i]->name, name) == 0) {

            return cur->childrenDir[i];

        }

    }

    return NULL;

}

 

// DFS 计算目录大小

int dfs(Node* root) {

    int sum = 0;

 

    // 子目录

    for (int i = 0; i < root->dirCount; i++) {

        sum += dfs(root->childrenDir[i]);

    }

 

    // 文件

    for (int i = 0; i < root->fileCount; i++) {

        sum += root->childrenFile[i]->size;

    }

 

    root->size = sum;

    return sum;

}

 

// 排序比较（冒泡）

void sortNodes(Node** arr, int n) {

    for (int i = 0; i < n; i++) {

        for (int j = i + 1; j < n; j++) {

            if (arr[i]->size < arr[j]->size ||

                (arr[i]->size == arr[j]->size &&

                 strcmp(arr[i]->name, arr[j]->name) > 0)) {

 

                Node* tmp = arr[i];

                arr[i] = arr[j];

                arr[j] = tmp;

            }

        }

    }

}

 

int main() {

    char path[1000];

    char fileLine[20000];

    char sizeLine[20000];

 

    // 输入

    fgets(path, sizeof(path), stdin);

    fgets(fileLine, sizeof(fileLine), stdin);

    fgets(sizeLine, sizeof(sizeLine), stdin);

 

    // 去掉换行

    path[strcspn(path, "\n")] = 0;

    fileLine[strcspn(fileLine, "\n")] = 0;

    sizeLine[strcspn(sizeLine, "\n")] = 0;

 

    // 切割 file

    char* fileList[1000];

    int fileCnt = 0;

    char* token = strtok(fileLine, " ");

    while (token) {

        fileList[fileCnt++] = token;

        token = strtok(NULL, " ");

    }

 

    // 切割 size

    int sizeList[1000];

    int sizeCnt = 0;

    token = strtok(sizeLine, " ");

    while (token) {

        sizeList[sizeCnt++] = atoi(token);

        token = strtok(NULL, " ");

    }

 

    Node* root = createNode("/");

 

    // 构建目录树

    for (int i = 0; i < fileCnt; i++) {

        char tmp[1000];

        strcpy(tmp, fileList[i]);

 

        char* parts[200];

        int cnt = 0;

 

        token = strtok(tmp, "/");

        while (token) {

            parts[cnt++] = token;

            token = strtok(NULL, "/");

        }

 

        Node* cur = root;

 

        // 从第1个开始（跳过根）

        for (int j = 1; j < cnt; j++) {

            char* name = parts[j];

            int isFile = (j == cnt - 1);

 

            if (!isFile) {

                Node* dir = findDir(cur, name);

                if (!dir) {

                    dir = createNode(name);

                    cur->childrenDir[cur->dirCount++] = dir;

                }

                cur = dir;

            } else {

                Node* fileNode = createNode(name);

                fileNode->size = sizeList[i];

                cur->childrenFile[cur->fileCount++] = fileNode;

            }

        }

    }

 

    // 找目标目录

    Node* cur = root;

    char pathCopy[1000];

    strcpy(pathCopy, path);

 

    char* parts[200];

    int cnt = 0;

 

    token = strtok(pathCopy, "/");

    while (token) {

        parts[cnt++] = token;

        token = strtok(NULL, "/");

    }

 

    for (int i = 1; i < cnt; i++) {

        Node* dir = findDir(cur, parts[i]);

        if (!dir) {

            return 0;

        }

        cur = dir;

    }

 

    // 计算大小

    dfs(cur);

 

    // 收集子节点

    Node* list[200];

    int total = 0;

 

    for (int i = 0; i < cur->dirCount; i++) {

        list[total++] = cur->childrenDir[i];

    }

    for (int i = 0; i < cur->fileCount; i++) {

        list[total++] = cur->childrenFile[i];

    }

 

    if (total == 0) return 0;

 

    // 排序

    sortNodes(list, total);

 

    int maxSize = list[0]->size;

 

    // 输出

    for (int i = 0; i < total; i++) {

        if (list[i]->size != maxSize) break;

        printf("%s", list[i]->name);

        if (i < total - 1 && list[i+1]->size == maxSize) {

            printf(" ");

        }

    }

 

    return 0;

}
