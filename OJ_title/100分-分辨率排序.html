100分-分辨率排序(C/C++/Py/Java/Js/Go)

题目描述

4K、2K、1080P、720P 清晰度定义如下（清晰度 = 宽 × 高）：

  • 4K：3840 × 2160
  • 2K：2560 × 1440
  • 1080P：1920 × 1080
  • 720P：1280 × 720

清晰度大小定义：720P < 1080P < 2K < 4K

分辨率（宽 × 高）匹配清晰度的规则：
给定任意分辨率（宽高），宽和高同时都大于等于某清晰度定义的宽、高时，才认为满足该清晰度定义，并且优先匹配高级别清晰度。

例如：2600 × 1400

  • 2600 < 3840，1400 < 2160，不满足 4K 清晰度定义
  • 2600 > 2560，1400 < 1440，不满足 2K 清晰度定义
  • 2600 > 1920，1400 > 1080，满足 1080P 清晰度定义

因此 2600 × 1400 是 1080P 清晰度。

特别说明：

  1. 所有低于 720P 的清晰度，都认为是 720P。
  2. 最大清晰度为 4K，只要满足 4K 清晰度标准的宽、高，无论多大，都归为 4K。
  3. 简化逻辑不考虑交换宽高的情况，例如 2500 × 3200，匹配为 1080P，不能匹配为 2K。

分辨率大小规则：

  1. 优先比较该分辨率对应的清晰度大小。
  2. 如果清晰度一致，则比较面积（宽 × 高）。
  3. 如果清晰度和面积一致，则比较“宽”。

输入描述

n 组“宽高”字符串，空格间隔，n < 10。

输出描述

从大到小排序后的 n 组“宽高”字符串，空格间隔。

用例1

输入

3840x2160

输出

3840x2160

题解

思路：模拟 + 自定义排序

  1. 对输入字符串按空格分割，得到每组分辨率规格。
  2. 对每组分辨率计算对应清晰度等级、面积。
  3. 按规则排序：
    • 先按清晰度降序
    • 清晰度相同按面积降序
    • 面积相同按宽降序
  4. 输出排序后的结果。

struct node{

// 通用 切割函数 函数 将字符串str根据delimiter进行切割

vector<string> split(const string& str, const string& delimiter) {

    vector<string> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(str.substr(start, end - start));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(str.substr(start));

    return result;

}

 

// 获取对应分辨率

int getLevel(int width, int height) {

    if (width >= 3840 && height >= 2160) return 4;

    if (width >= 2560 && height >= 1440) return 3;

    if (width >= 1920 && height >= 1080) return 2;

    return 1;

}

 

string resolutionSort(string& input) {

    vector<string> items= split(input, " ");

    int n = items.size();

    vector<node> nodes;

    for (int i = 0; i < n; i++) {

        string item = items[i];

        vector<string> tmp = split(item, "x");

        int width = stoi(tmp[0]);

        int height = stoi(tmp[1]);

        int level = getLevel(width, height);

        int area = width * height;

        nodes.push_back({width, height, level, area});

    }

 

    // 自定义排序

    sort(nodes.begin(), nodes.end(), [](node& a, node& b) {

        if (a.level != b.level) {

            return a.level > b.level;

        }

        if (a.area != b.area) {

            return a.area > b.area;

        }

        return a.width > b.width;

    });

 

    // 构建结果字符串

    string res;

    for (int i = 0; i < nodes.size(); i++) {

        node a = nodes[i];

        int width = a.width;

        int height = a.height;

        res += (to_string(width) + "x" + to_string(height));

        if (i != nodes.size() -1) {

            res += " ";

        }

    }

    return res;

}

 

 

int main() {

    string input;

    getline(cin, input);

    string res = resolutionSort(input);

    cout << res;

    return 0;

}

import java.util.*;

 

class Node {

    int width;

    int height;

    int level;

    int area;

 

    Node(int width, int height, int level, int area) {

        this.width = width;

        this.height = height;

        this.level = level;

        this.area = area;

    }

}

 

public class Main {

 

    // 获取对应分辨率等级

    public static int getLevel(int width, int height) {

        if (width >= 3840 && height >= 2160) return 4;

        if (width >= 2560 && height >= 1440) return 3;

        if (width >= 1920 && height >= 1080) return 2;

        return 1;

    }

 

    public static String resolutionSort(String input) {

        String[] items = input.split(" ");

        List<Node> nodes = new ArrayList<>();

 

        for (String item : items) {

            String[] tmp = item.split("x");

            int width = Integer.parseInt(tmp[0]);

            int height = Integer.parseInt(tmp[1]);

            int level = getLevel(width, height);

            int area = width * height;

 

            nodes.add(new Node(width, height, level, area));

        }

 

        // 自定义排序

        nodes.sort((a, b) -> {

            if (a.level != b.level) return b.level - a.level;

            if (a.area != b.area) return b.area - a.area;

            return b.width - a.width;

        });

 

        // 构建结果字符串

        StringBuilder res = new StringBuilder();

        for (int i = 0; i < nodes.size(); i++) {

            Node a = nodes.get(i);

            res.append(a.width).append("x").append(a.height);

            if (i != nodes.size() - 1) {

                res.append(" ");

            }

        }

 

        return res.toString();

    }

 

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        String input = sc.nextLine();

 

        System.out.println(resolutionSort(input));

    }

}

package main

 

import (

       "bufio"

       "fmt"

       "os"

       "sort"

       "strconv"

       "strings"

)

 

type Node struct {

       width  int

       height int

       level  int

       area   int

}

 

// 获取对应分辨率等级

func getLevel(width, height int) int {

       if width >= 3840 && height >= 2160 {

              return 4

       }

       if width >= 2560 && height >= 1440 {

              return 3

       }

       if width >= 1920 && height >= 1080 {

              return 2

       }

       return 1

}

 

func resolutionSort(input string) string {

       items := strings.Split(input, " ")

       nodes := []Node{}

 

       for _, item := range items {

              parts := strings.Split(item, "x")

              width, _ := strconv.Atoi(parts[0])

              height, _ := strconv.Atoi(parts[1])

 

              level := getLevel(width, height)

              area := width * height

 

              nodes = append(nodes, Node{width, height, level, area})

       }

 

       // 自定义排序

       sort.Slice(nodes, func(i, j int) bool {

              if nodes[i].level != nodes[j].level {

                     return nodes[i].level > nodes[j].level

              }

              if nodes[i].area != nodes[j].area {

                     return nodes[i].area > nodes[j].area

              }

              return nodes[i].width > nodes[j].width

       })

 

       // 构建结果字符串

       res := []string{}

       for _, n := range nodes {

              res = append(res, fmt.Sprintf("%dx%d", n.width, n.height))

       }

 

       return strings.Join(res, " ")

}

 

func main() {

       reader := bufio.NewReader(os.Stdin)

       input, _ := reader.ReadString('\n')

       input = strings.TrimSpace(input)

 

       fmt.Println(resolutionSort(input))

}

 

C语言

#include <stdio.h>

#include <stdlib.h>

#include <string.h>

 

typedef struct {

    int width;

    int height;

    int level;

    int area;

} Node;

 

// 获取对应分辨率等级

int getLevel(int width, int height) {

    if (width >= 3840 && height >= 2160) return 4;

    if (width >= 2560 && height >= 1440) return 3;

    if (width >= 1920 && height >= 1080) return 2;

    return 1;

}

 

// 排序规则

int cmp(const void* a, const void* b) {

    Node* x = (Node*)a;

    Node* y = (Node*)b;

 

    if (x->level != y->level) return y->level - x->level;

    if (x->area != y->area) return y->area - x->area;

    return y->width - x->width;

}

 

 

void resolutionSort(char* input, char* output) {

    Node nodes[10];

    int count = 0;

 

    // 按空格切割

    char* token = strtok(input, " ");

    while (token != NULL) {

        int width, height;

        sscanf(token, "%dx%d", &width, &height);

 

        nodes[count].width = width;

        nodes[count].height = height;

        nodes[count].level = getLevel(width, height);

        nodes[count].area = width * height;

        count++;

 

        token = strtok(NULL, " ");

    }

 

    // 排序

    qsort(nodes, count, sizeof(Node), cmp);

 

    // 组装结果字符串

    char buffer[10000] = {0};

    char temp[50];

 

    for (int i = 0; i < count; i++) {

        sprintf(temp, "%dx%d", nodes[i].width, nodes[i].height);

        strcat(buffer, temp);

 

        if (i != count - 1) {

            strcat(buffer, " ");

        }

    }

 

    // 输出写入

    strcpy(output, buffer);

}

 

int main() {

    char input[1000];

    char output[1000];

 

    fgets(input, sizeof(input), stdin);

 

    // 去掉换行符

    input[strcspn(input, "\n")] = 0;

 

    resolutionSort(input, output);

 

    printf("%s\n", output);

 

    return 0;

}
