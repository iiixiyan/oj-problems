200分-小猫钓鱼纸牌游戏(C/C++/Py/Java/Js/Go)

题目描述

有两名玩家甲和乙，他们玩一种“小猫钓鱼”扑克牌游戏。扑克牌为 A、2、3、…、J、Q、K，不考虑花色，都用 1~13 表示。然后给甲、乙两人各发 n 张牌，按给定顺序排成队列，这些牌背面朝上、正面朝下放置，队列的第一个元素是背面的第一张牌。

游戏规则如下：

  1. 出牌与收牌
    • 初始由甲先出牌，打出的牌正面朝上放在桌面的最底部；然后乙出牌，正面朝上放在甲的牌上面；之后甲乙轮流出牌，直到触发收牌或者任意一方牌出完。
    • 如果玩家出牌后触发了收牌，则把收到的一摞牌作为整体翻面后，背面朝上、正面朝下放到现有手牌的底部，然后当前收牌的玩家继续出牌。
  2. 收牌规则
    • 若当前打出的牌点数，与桌面上之前某张牌的点数相同，则触发收牌。收牌范围是这两张相同点数牌之间的所有牌，含这两张。
    • 若当前打出的牌是 J，并且此时桌面上已有至少一张牌（不含当前这张），则当前玩家也会触发收牌，收牌范围是桌面上的所有牌。
    • 若桌面上原本没有牌，则 J 仅作为普通牌放到桌面。
  3. 游戏结束条件
    • 若某位玩家在自己回合开始时已经没有牌可出，则游戏立刻结束。若此时对方还有余牌，则对方获胜；若对方也无牌，则平局。
    • 若在模拟过程中，出牌总次数超过上限 10000 次仍未结束，则认为进入死循环，判定为平局。

输入描述

输入为甲的初始牌队列和乙的初始牌队列。
甲乙初始手牌数量相等，均为整数 n，且 1 <= n <= 100。

输出描述

  • 一方获胜时，输出获胜方手中背面朝上最上方的那张牌的数值。
  • 若平局：
    • 桌面无牌则输出 0
    • 否则输出当前桌面上正面朝上的最上方那张牌的数值

用例1

输入

1,2

10,12

输出

12

说明
甲的牌堆为 1,2
乙的牌堆为 10,12

出牌顺序：甲出 1，乙出 10，甲出 2，乙出 12，此时甲无牌可出，乙也无余牌，游戏结束，判为平局。
桌面牌堆为 1,10,2,12，平局输出桌面正面朝上最上方的牌值 12。

用例2

输入

1,3

1,3

输出

1

说明
甲的牌堆为 1,3
乙的牌堆为 1,3

一种模拟过程如下：

  1. 甲出 1，剩余 3，桌面为 1
  2. 乙出 1，匹配桌面 1，收走 1,1，桌面变空，乙变为 3,1,1
  3. 乙出 3，剩余 1,1，桌面为 3
  4. 甲出 3，匹配桌面 3，收走 3,3，桌面变空，甲变为 3,3
  5. 甲出 3，剩余 3，桌面为 3
  6. 乙出 1，剩余 1，桌面为 3,1
  7. 甲出 3，匹配桌面 3，收走 3,1,3，桌面变空，甲变为 3,1,3
  8. 甲出 3，剩余 1,3，桌面为 3
  9. 乙出 1，乙变空，桌面变为 3,1
  10. 甲出 1，匹配桌面 1，收走 1,1，桌面变为 3，甲变为 3,1,1
  11. 甲出 3，匹配桌面 3，收走 3,3，桌面变空，甲变成 1,1,3,3
  12. 甲出 1，变为 1,3,3，桌面为 1
  13. 乙无牌可出，甲获胜，甲手牌非空，输出背面最上方的牌 1

题解

思路：纯模拟

  1. 用双端队列维护甲乙当前手牌，桌面牌用数组或栈模拟。
  2. 再用一个位置数组 pos[14] 记录某个点数最近一次在桌面中出现的位置，不存在记为 -1，用于快速判断是否触发收牌。
  3. 每轮出牌时：
    • 若当前玩家无牌可出，直接根据双方和桌面状态返回结果。
    • 否则打出队首牌，压到桌面顶部。
    • 判断是否触发收牌：
      • 若当前牌是 11(J) 且桌面原本非空，则收走桌面全部牌。
      • 否则若桌面之前已出现过相同点数，则收走从上次出现位置到当前顶部的所有牌。
  4. 收牌后，将这一摞牌整体翻面并放到当前玩家手牌队尾，且当前玩家继续出牌；未收牌则轮到对方。
  5. 若总出牌次数超过 10000，判为平局：
    • 桌面空，输出 0
    • 否则输出桌面最上方牌值。

vector<int> split(const string& str, const string& delimiter) {

    vector<int> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(stoi(str.substr(start, end - start)));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(stoi(str.substr(start)));

    return result;

}

 

int fishGame(vector<int>& aCard, vector<int> & bCard) {

    deque<int> a(aCard.begin(), aCard.end());

    deque<int> b(bCard.begin(), bCard.end());

    // 牌桌上的牌

    vector<int> tableCard;

 

    // 0 表示甲出牌 1表示乙出牌

    int current = 0;

 

    // 维护牌面值在牌桌上出现的位置

    vector<int> pos(14, -1);

 

 

    for (int i = 0; i <= 10000; i++) {

        deque<int>& currentCard = (current == 0) ? a : b;

 

        deque<int>& otherCard = (current == 0) ? b : a;

 

        // 当前出牌牌堆为空,游戏结束

        if (currentCard.empty()) {

            // 对手有余牌

            if (!otherCard.empty()) {

               

                return otherCard.front();

            } else {

               

                // 根据桌面牌堆输出

                return tableCard.empty() ? 0 : tableCard.back();

            }

        }

       

        int val = currentCard.front();

 

        currentCard.pop_front();

       

        int samePos= pos[val];

        bool tableEmptyFlag = tableCard.empty();

 

        // 放置当前牌到牌堆

        tableCard.push_back(val);

        pos[val] = tableCard.size() - 1;

        // 是否能收牌

        bool takeUpFlag = false;

        // 收牌起始位置

        int takeUpStart = -1;

        // J且桌面牌非空

        if (val == 11 && !tableEmptyFlag) {

            takeUpFlag = true;

            takeUpStart = 0;

        } else if (samePos != -1) {

            takeUpFlag = true;

            takeUpStart =  samePos;

        }

        // 可以收牌

        if (takeUpFlag) {

            vector<int> takeUpCard(tableCard.begin() + takeUpStart, tableCard.end());

            tableCard.resize(takeUpStart);

           

            // 移除位置和加入牌堆

            for (auto v : takeUpCard) {

                pos[v] = -1;

                currentCard.push_back(v);

            }

           

        // 不可以收牌，切换用户

        } else {

            current ^= 1;

        }

    }

 

    // 超过10000且桌面牌非空

    if (!tableCard.empty()) {

        return tableCard.back();;

    }

   

    // 空返回0

    return 0;

}

                takeUpStart = 0;

                takeUpFlag = true;

                takeUpStart = samePos;

            }

                List<Integer> takeUpCard = tableCard.subList(takeUpStart, tableCard.size());

                List<Integer> copy = new ArrayList<>(takeUpCard);

                    currentCard.addLast(v);

                }

            } else {

                current ^= 1;

            }

        }

    }

func split(s string, delimiter string) []int {

    parts := strings.Split(s, delimiter)

    res := make([]int, len(parts))

    for i, v := range parts {

        res[i], _ = strconv.Atoi(v)

    }

    return res

func fishGame(aCard []int, bCard []int) int {

    a := append([]int{}, aCard...)

    b := append([]int{}, bCard...)

 

    // 牌桌上的牌

    tableCard := []int{}

 

    // 0 表示甲出牌 1表示乙出牌

    current := 0

 

    // 维护牌面值在牌桌上出现的位置

    pos := make([]int, 14)

    for i := range pos {

        pos[i] = -1

    }

 

    for i := 0; i <= 10000; i++ {

 

        var currentCard *[]int

        var otherCard *[]int

 

        if current == 0 {

            currentCard = &a

            otherCard = &b

        } else {

            currentCard = &b

            otherCard = &a

        }

 

        // 当前出牌牌堆为空,游戏结束

        if len(*currentCard) == 0 {

            // 对手有余牌

            if len(*otherCard) > 0 {

                return (*otherCard)[0]

            } else {

                // 根据桌面牌堆输出

                if len(tableCard) > 0 {

                    return tableCard[len(tableCard)-1]

                }

                return 0

            }

        }

 

        val := (*currentCard)[0]

        *currentCard = (*currentCard)[1:]

 

        samePos := pos[val]

        tableEmptyFlag := len(tableCard) == 0

 

        // 放置当前牌到牌堆

        tableCard = append(tableCard, val)

        pos[val] = len(tableCard) - 1

 

        // 是否能收牌

        takeUpFlag := false

        takeUpStart := -1

 

        // J且桌面牌非空

        if val == 11 && !tableEmptyFlag {

            takeUpFlag = true

            takeUpStart = 0

        } else if samePos != -1 {

            takeUpFlag = true

            takeUpStart = samePos

        }

 

        // 可以收牌

        if takeUpFlag {

            takeUpCard := append([]int{}, tableCard[takeUpStart:]...)

            tableCard = tableCard[:takeUpStart]

 

            // 移除位置和加入牌堆

            for _, v := range takeUpCard {

                pos[v] = -1

                *currentCard = append(*currentCard, v)

            }

        } else {

            // 不可以收牌，切换用户

            current ^= 1

        }

    }

 

    // 超过10000且桌面牌非空

    if len(tableCard) > 0 {

        return tableCard[len(tableCard)-1]

    }

 

    // 空返回0

}

int split(char *str, int arr[]) {

    int count = 0;

    char *token = strtok(str, ",");

    while (token) {

        arr[count++] = atoi(token);

        token = strtok(NULL, ",");

    }

    return count;

}

 

int fishGame(int a[], int aSize, int b[], int bSize) {

 

    // 牌桌上的牌

    int table[10000], tableSize = 0;

 

    // 维护牌面值在牌桌上出现的位置

    int pos[14];

    for (int i = 0; i < 14; i++) pos[i] = -1;

 

    // 0 表示甲出牌 1表示乙出牌

    int current = 0;

 

    for (int step = 0; step <= 10000; step++) {

 

        int *currentCard = current == 0 ? a : b;

        int *otherCard = current == 0 ? b : a;

        int *currentSize = current == 0 ? &aSize : &bSize;

        int *otherSize = current == 0 ? &bSize : &aSize;

 

        // 当前出牌牌堆为空,游戏结束

        if (*currentSize == 0) {

            // 对手有余牌

            if (*otherSize > 0) return otherCard[0];

            else return tableSize ? table[tableSize - 1] : 0;

        }

 

        int val = currentCard[0];

 

        // 出队

        for (int i = 1; i < *currentSize; i++)

            currentCard[i - 1] = currentCard[i];

        (*currentSize)--;

 

        int samePos = pos[val];

        int tableEmptyFlag = tableSize == 0;

 

        // 放置当前牌到牌堆

        table[tableSize++] = val;

        pos[val] = tableSize - 1;

 

        // 是否能收牌

        int takeUpFlag = 0;

        int takeUpStart = -1;

 

        // J且桌面牌非空

        if (val == 11 && !tableEmptyFlag) {

            takeUpFlag = 1;

            takeUpStart = 0;

        } else if (samePos != -1) {

            takeUpFlag = 1;

            takeUpStart = samePos;

        }

 

        // 可以收牌

        if (takeUpFlag) {

            int takeSize = tableSize - takeUpStart;

 

            // 移除位置和加入牌堆

            for (int i = 0; i < takeSize; i++) {

                int v = table[takeUpStart + i];

                pos[v] = -1;

                currentCard[(*currentSize)++] = v;

            }

 

            tableSize = takeUpStart;

        } else {

            // 不可以收牌，切换用户

            current ^= 1;

        }

    }

 

    // 超过10000且桌面牌非空

    if (tableSize > 0) return table[tableSize - 1];

 

    // 空返回0

    return 0;

}
