200分-WIFI设备网络规划(C/C++/Py/Java/Js/Go)

题目描述

WIFI 网络中，专业的网络规划不仅可以提升业务体验，还可以减少部署成本。把办公区可以看作一个 m * n 的网格，部分网格包含墙壁（无法放置 AP），部分为空地（可以放置 AP）。每个 AP 覆盖范围是一个 3*3 的正方形（包括自身位置、上下左右以及对角线区域），且 AP 和 AP 的覆盖区域不能重叠，防止相互干扰。

现在给定一个 m x n（不超过 50 * 50）的网络布局图（墙壁用字符 # 表示，空地用字符 . 表示），请设计一个算法，计算最少放置多少数量的 AP 来覆盖所有空地；如果不能按条件完成覆盖，请返回 -1。

输入描述

第一行输入 m n

接下来 m 行输入，每行一个字符串

输出描述

最少放置多少数量的 AP 来覆盖所有空地；如果不能按条件完成覆盖，请返回 -1。

用例1

输入

3 3

#.#

#.#

#.#

输出

1

用例2

输入

4 3

#.#

#.#

#.#

#.#

输出

2

题解

思路：递归回溯

  1. 预处理
  • 每个空地都可以看作 AP 可放置候选集，预处理统计每个 AP 能够覆盖的空地以及覆盖区域
  • 利用 AP 候选信息，统计出每个空地能被覆盖的 AP 信息，用于后续剪枝。使用 pointToAP 存储
  1. 递归回溯计算最少 AP 数
  • 创建 covered 布尔数组，跟踪指定编号空地是否被覆盖
  • 创建 used[][] 布尔数组表示指定位置是否被覆盖，用于快速判断 AP 放置是否会产生冲突
  • 定义 ans 表示能够覆盖所有空地的最少 AP 数，初始可设为一个很大值
  • 然后进行正常的递归回溯 + 剪枝搜索，主要剪枝思路：
    • 如果当前所用 AP 数已超过 ans，直接剪枝
    • 如果 当前使用AP数 + 未覆盖空地数 / 9 >= ans，直接剪枝
    • 如果当前局面存在某个空地不能再被任何合法 AP 覆盖，直接剪枝
    • 优先处理当前可覆盖 AP 最少的空地作为扩展点，利用 pointToAP 找出符合条件的 AP 候选位置，并优先按可覆盖空地数多的点进行递归
  1. 根据第 2 步结果
  • 如果最终 ans 仍不可达，返回 -1
  • 否则返回 ans

C++

#include<iostream>

#include<vector>

#include<string>

#include <utility>

#include <sstream>

#include<algorithm>

#include<cstring>

#include<climits>

#include<map>

using namespace std;

 

const int MAXN = 55;

const int MAXP = 2500;

 

int n, m;

string g[MAXN];

 

int id[MAXN][MAXN];

int tot = 0;

 

struct AP {

    vector<int> dots;                  // 覆盖的'.'

    vector<pair<int,int>> cells;       // 3x3占用格子

};

 

vector<AP> apList;

 

// 空地能够被覆盖的AP的编号

vector<vector<int>> pointToAP;

// 空地是否被覆盖

bool covered[MAXP];

// 已被覆盖点

bool used[MAXN][MAXN];

 

int ans;

 

// 方向

int dx[9]={-1,-1,-1,0,0,0,1,1,1};

int dy[9]={-1,0,1,-1,0,1,-1,0,1};

 

bool in(int x,int y){

    return x>=0 && x<n && y>=0 && y<m;

}

 

// 预处理候选集AP信息 覆盖的空地位置/ 占用格子位置

void buildAP(){

    for(int i=0;i<n;i++){

        for(int j=0;j<m;j++){

            if(g[i][j] != '.') continue;

 

            AP ap;

 

            for(int k=0;k<9;k++){

                int x=i+dx[k], y=j+dy[k];

                if(!in(x,y)) continue;

 

                ap.cells.push_back({x,y});

 

                if(g[x][y]=='.'){

                    ap.dots.push_back(id[x][y]);

                }

            }

 

            apList.push_back(ap);

        }

    }

}

 

// 剩余未覆盖点

int remain(){

    int cnt=0;

    for(int i=0;i<tot;i++)

        if(!covered[i]) cnt++;

    return cnt;

}

 

// 选取最佳点

int selectPoint(){

    int best=-1, bestCnt=1e9;

 

    for(int i=0;i<tot;i++){

        if(covered[i]) continue;

 

        int cnt=0;

        for(int ap:pointToAP[i]){

            bool ok=true;

            for(auto &c:apList[ap].cells){

                if(used[c.first][c.second]){

                    ok=false; break;

                }

            }

            if(ok) cnt++;

        }

        // 存在某个点不能被任何AP覆盖，直接返回-1

        if(cnt==0) return -1;

 

        if(cnt < bestCnt){

            bestCnt = cnt;

            best = i;

        }

    }

 

    return best;

}

 

// 递归回溯

void dfs(int usedCnt){

    if(usedCnt >= ans) return;

   

    // 判断是否已全部覆盖

    bool all = true;

    for(int i=0;i<tot;i++){

        if(!covered[i]){

            all=false; break;

        }

    }

    if(all){

        ans = usedCnt;

        return;

    }

    // 理论最少还需要多少AP，超过最佳直接剪枝

    int lb = (remain() + 8) / 9;

    if(usedCnt + lb >= ans) return;

   

    // 选取最佳点，可覆盖AP最少的空地作为扩展点

    int p = selectPoint();

    if(p == -1) return;

 

    vector<int> cand;

 

    for(int ap: pointToAP[p]){

        bool ok=true;

        for(auto &c: apList[ap].cells){

            if(used[c.first][c.second]){

                ok=false; break;

            }

        }

        if(ok) cand.push_back(ap);

    }

   

    // 优先选取能覆盖位置多的AP

    sort(cand.begin(), cand.end(),

         [&](int a,int b){

            return apList[a].dots.size() > apList[b].dots.size();

         });

   

    // 递归回溯

    for(int ap: cand){

 

        vector<pair<int,int>> oldCells;

        vector<int> oldDots;

 

        // 放AP

        for(auto &c: apList[ap].cells){

            if(!used[c.first][c.second]){

                used[c.first][c.second] = true;

                oldCells.push_back(c);

            }

        }

 

        for(int d: apList[ap].dots){

            if(!covered[d]){

                covered[d] = true;

                oldDots.push_back(d);

            }

        }

 

        dfs(usedCnt + 1);

 

        // 回溯

        for(auto &c: oldCells){

            used[c.first][c.second] = false;

        }

 

        for(int d: oldDots){

            covered[d] = false;

        }

    }

}

 

int solve(vector<string>& grid){

    n = grid.size();

    m = grid[0].size();

 

    for(int i=0;i<n;i++) g[i] = grid[i];

 

    memset(id, -1, sizeof(id));

    memset(covered, 0, sizeof(covered));

    memset(used, 0, sizeof(used));

 

    tot = 0;

    apList.clear();

    pointToAP.clear();

 

    // 给空地编号 '.'

    for(int i=0;i<n;i++){

        for(int j=0;j<m;j++){

            if(g[i][j]=='.'){

                id[i][j] = tot++;

            }

        }

    }

    // 不存在空地直接返回0

    if(tot == 0) return 0;

 

    pointToAP.resize(tot);

    // 预处理

    buildAP();

 

    // 反向索引

    for(int i=0;i<apList.size();i++){

        for(int d: apList[i].dots){

            pointToAP[d].push_back(i);

        }

    }

 

    ans = INT_MAX;

 

    dfs(0);

 

    return (ans == INT_MAX ? -1 : ans);

}

 

int main(){

    cin >> n >> m;

    vector<string> grid(n);

 

    for(int i=0;i<n;i++){

        cin >> grid[i];

    }

 

    cout << solve(grid) << endl;

    return 0;

}

import sys

sys.setrecursionlimit(1000000)

 

# 输入

n, m = map(int, input().split())

g = [list(input().strip()) for _ in range(n)]

 

# 空地编号

id_map = [[-1]*m for _ in range(n)]

tot = 0

 

# AP结构：覆盖点 + 3x3占用格子

class AP:

    def __init__(self):

        self.dots = []   # 覆盖的'.'

        self.cells = []  # 3x3占用格子

 

ap_list = []

 

# 点 -> AP

point_to_ap = []

 

# 是否覆盖

covered = []

 

# 是否占用

used = [[False]*m for _ in range(n)]

 

ans = float('inf')

 

dx = [-1,-1,-1,0,0,0,1,1,1]

dy = [-1,0,1,-1,0,1,-1,0,1]

 

def inb(x,y):

    return 0 <= x < n and 0 <= y < m

 

# 构建AP

def build_ap():

    for i in range(n):

        for j in range(m):

            if g[i][j] != '.':

                continue

 

            ap = AP()

 

            for k in range(9):

                x, y = i + dx[k], j + dy[k]

                if not inb(x,y):

                    continue

 

                ap.cells.append((x,y))

 

                # 覆盖空地

                if g[x][y] == '.':

                    ap.dots.append(id_map[x][y])

 

            ap_list.append(ap)

 

# 剩余未覆盖点

def remain():

    return sum(not x for x in covered)

 

# MRV选点

def select_point():

    best = -1

    best_cnt = 10**9

 

    for i in range(tot):

        if covered[i]:

            continue

 

        cnt = 0

 

        for ap in point_to_ap[i]:

            ok = True

 

            # 检查冲突

            for x,y in ap_list[ap].cells:

                if used[x][y]:

                    ok = False

                    break

 

            if ok:

                cnt += 1

 

        if cnt == 0:

            return -1

 

        if cnt < best_cnt:

            best_cnt = cnt

            best = i

 

    return best

 

# DFS回溯

def dfs(cnt):

    global ans

 

    if cnt >= ans:

        return

 

    # 是否全部覆盖

    if all(covered):

        ans = cnt

        return

 

    # 下界剪枝

    if cnt + (remain()+8)//9 >= ans:

        return

 

    p = select_point()

    if p == -1:

        return

 

    cand = []

 

    for ap in point_to_ap[p]:

        ok = True

 

        for x,y in ap_list[ap].cells:

            if used[x][y]:

                ok = False

                break

 

        if ok:

            cand.append(ap)

 

    # 优先覆盖多的AP

    cand.sort(key=lambda x: len(ap_list[x].dots), reverse=True)

 

    for ap in cand:

 

        old_cells = []

        old_dots = []

 

        # 放AP

        for x,y in ap_list[ap].cells:

            if not used[x][y]:

                used[x][y] = True

                old_cells.append((x,y))

 

        for d in ap_list[ap].dots:

            if not covered[d]:

                covered[d] = True

                old_dots.append(d)

 

        dfs(cnt + 1)

 

        # 回溯

        for x,y in old_cells:

            used[x][y] = False

 

        for d in old_dots:

            covered[d] = False

 

# solve

def solve():

    global tot, covered, point_to_ap

 

    # 编号空地

    for i in range(n):

        for j in range(m):

            if g[i][j] == '.':

                id_map[i][j] = tot

                tot += 1

 

    if tot == 0:

        return 0

 

    covered = [False] * tot

    point_to_ap = [[] for _ in range(tot)]

 

    build_ap()

 

    # 点 -> AP

    for i, ap in enumerate(ap_list):

        for d in ap.dots:

            point_to_ap[d].append(i)

 

    dfs(0)

 

    return -1 if ans == float('inf') else ans

 

print(solve())

                break;

            ans = usedCnt;

            return;

        }

 

        // 理论最少还需要多少AP，超过最佳直接剪枝

        let lb = Math.ceil(remain() / 9);

        if (usedCnt + lb >= ans) return;

        if (p == -1) return;

            let ok = true;

                if (used[x][y]) {

                    break;

                }

            }

 

            if (ok) cand.push(ap);

        }

 

        // 优先选取能覆盖位置多的AP

        cand.sort((a, b) =>

            apList[b].dots.length - apList[a].dots.length

        );

 

        for (let ap of cand) {

 

            let oldCells = [];

            let oldDots = [];

 

            for (let [x, y] of apList[ap].cells) {

                if (!used[x][y]) {

                    used[x][y] = true;

                    oldCells.push([x, y]);

                }

            }

 

            for (let d of apList[ap].dots) {

                if (!covered[d]) {

                    covered[d] = true;

                    oldDots.push(d);

                }

            }

 

            dfs(usedCnt + 1);

 

            for (let [x, y] of oldCells) {

                used[x][y] = false;

            }

 

            for (let d of oldDots) {

                covered[d] = false;

            }

        }

    }

        // 空地编号

        for (let i = 0; i < n; i++) {

                if (g[i][j] == '.') {

                }

            }

       // 预处理

         // 反向构建索引

        for (let i = 0; i < apList.length; i++) {

            for (let d of apList[i].dots) {

        }

#include <stdio.h>

#include <string.h>

 

#define MAXN 55

#define MAXP 2500

 

// AP结构

typedef struct {

    int dots[MAXP];

    int dotSize;

    int cells[9][2];

    int cellSize;

} AP;

 

int n,m;

char g[MAXN][MAXN];

 

int id[MAXN][MAXN];

int tot;

 

// AP列表

AP apList[MAXP];

int apCnt;

 

// 空地能够被覆盖的AP的编号

int pointToAP[MAXP][MAXP];

int ptSize[MAXP];

 

// 空地是否被覆盖

int covered[MAXP];

 

// 已被覆盖点

int used[MAXN][MAXN];

 

int ans = 1e9;

 

// 方向

int dx[9]={-1,-1,-1,0,0,0,1,1,1};

int dy[9]={-1,0,1,-1,0,1,-1,0,1};

 

int in(int x,int y){

    return x>=0 && x<n && y>=0 && y<m;

}

 

// 预处理候选集AP信息 覆盖的空地位置/ 占用格子位置

void buildAP(){

    for(int i=0;i<n;i++){

        for(int j=0;j<m;j++){

            if(g[i][j] != '.') continue;

 

            AP ap;

            ap.dotSize=0;

            ap.cellSize=0;

 

            for(int k=0;k<9;k++){

                int x=i+dx[k], y=j+dy[k];

                if(!in(x,y)) continue;

 

                ap.cells[ap.cellSize][0]=x;

                ap.cells[ap.cellSize][1]=y;

                ap.cellSize++;

 

                if(g[x][y]=='.'){

                    ap.dots[ap.dotSize++]=id[x][y];

                }

            }

 

            apList[apCnt++]=ap;

        }

    }

}

 

// 剩余未覆盖点

int remain(){

    int c=0;

    for(int i=0;i<tot;i++)

        if(!covered[i]) c++;

    return c;

}

 

// 选取最佳点

int selectPoint(){

    int best=-1;

    int bestCnt=1e9;

 

    for(int i=0;i<tot;i++){

        if(covered[i]) continue;

 

        int cnt=0;

 

        for(int k=0;k<ptSize[i];k++){

            int ap=pointToAP[i][k];

 

            int ok=1;

 

            for(int j=0;j<apList[ap].cellSize;j++){

                int x=apList[ap].cells[j][0];

                int y=apList[ap].cells[j][1];

 

                if(used[x][y]){

                    ok=0;

                    break;

                }

            }

 

            if(ok) cnt++;

        }

 

        if(cnt==0) return -1;

 

        if(cnt<bestCnt){

            bestCnt=cnt;

            best=i;

        }

    }

 

    return best;

}

 

// 递归回溯

void dfs(int usedCnt){

 

    if(usedCnt>=ans) return;

 

    // 判断是否已全部覆盖

    int all=1;

    for(int i=0;i<tot;i++){

        if(!covered[i]){

            all=0; break;

        }

    }

 

    if(all){

        ans=usedCnt;

        return;

    }

 

    // 理论最少还需要多少AP，超过最佳直接剪枝

    int lb=(remain()+8)/9;

    if(usedCnt+lb>=ans) return;

 

    int p=selectPoint();

    if(p==-1) return;

 

    int cand[MAXP],cs=0;

 

    for(int i=0;i<ptSize[p];i++){

        int ap=pointToAP[p][i];

 

        int ok=1;

 

        for(int j=0;j<apList[ap].cellSize;j++){

            int x=apList[ap].cells[j][0];

            int y=apList[ap].cells[j][1];

 

            if(used[x][y]){

                ok=0;

                break;

            }

        }

 

        if(ok) cand[cs++]=ap;

    }

 

    for(int i=0;i<cs;i++){

        for(int j=i+1;j<cs;j++){

            if(apList[cand[j]].dotSize > apList[cand[i]].dotSize){

                int t=cand[i];

                cand[i]=cand[j];

                cand[j]=t;

            }

        }

    }

 

    for(int i=0;i<cs;i++){

 

        int ap=cand[i];

 

        int oldCells[9][2],oc=0;

        int oldDots[MAXP],od=0;

 

        for(int j=0;j<apList[ap].cellSize;j++){

            int x=apList[ap].cells[j][0];

            int y=apList[ap].cells[j][1];

 

            if(!used[x][y]){

                used[x][y]=1;

                oldCells[oc][0]=x;

                oldCells[oc][1]=y;

                oc++;

            }

        }

 

        for(int j=0;j<apList[ap].dotSize;j++){

            int d=apList[ap].dots[j];

            if(!covered[d]){

                covered[d]=1;

                oldDots[od++]=d;

            }

        }

 

        dfs(usedCnt+1);

 

        for(int j=0;j<oc;j++){

            used[oldCells[j][0]][oldCells[j][1]]=0;

        }

 

        for(int j=0;j<od;j++){

            covered[oldDots[j]]=0;

        }

    }

}
