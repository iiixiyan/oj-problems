100分 计算数列位置N的值

题目描述：

1.     输入 M、N 两个数，按照如下规则构建一个数列。

2.     数列的前 M 个元素的值为 1 到 M。

3.     从第 M+1 个元素开始，计算的逻辑为：

o    如果其前面的 M 个元素中，存在值相同的元素，则该位置上的数值等于前面 M 个数中最大的数值与最小的数值之和；

o    如果其前面的 M 个元素中，不存在值相同的元素，则该位置上的数值等于前面 M 个数中最大的数值与最小的数值之差；

请计算该数列第 N 个位置上的数值。

补充说明：
M 取值范围：3 <= M <= 10
N 取值范围：1 <= N <= 50

输入描述

输出N和M，使用，分割

输出描述

输出N位置上的数值

用例1

输入

5,1

输出

1

 

思路：逻辑分析

1.  此题数据量比较小，暴力递推所有前 n 个位置的值。

2.  对于 n <= m 的情况可以直接，返回 n。

3.  对于 n > m 的情况，对于 m+1 及之后的位置可直接通过模拟，判断前 n 个数是否存在重复值，并记录最大值、最小值。从而确认当前位置值。

int positionValue(int m , int n) {

    // 小于情况就是n

    if (n <= m) {

        return n;

    }

    vector<int> num(n + 1);

    // 初始化

    for (int i = 1; i <= m; i++) {

        num[i] = i;

    }

    // 数据量小->递推

    for (int i = m + 1; i <= n; i++) {

        int mx = INT_MIN, mn = INT_MAX;

        set<int> s;

        bool dupFlag = false;

        for (int j = i - m; j < i; j++) {

            if (s.count(num[j])) {

                dupFlag = true;

            }

            s.insert(num[j]);

 

            mx = max(mx, num[j]);

            mn = min(mn, num[j]);

        }

        if (dupFlag) {

            num[i] = mx + mn;

        } else {

            num[i] = mx - mn;

        }

    }

    return num[n];

}

    scanf("%d,%d", &m, &n);

    int res = positionValue(m, n);

    cout << res;

    return 0;

}

#include <stdio.h>

#include <limits.h>

 

int positionValue(int m, int n) {

    // 小于情况就是n

    if (n <= m) {

        return n;

    }

 

    int num[55] = {0}; // n <= 50

 

    // 初始化

    for (int i = 1; i <= m; i++) {

        num[i] = i;

    }

 

    // 数据量小->递推

    for (int i = m + 1; i <= n; i++) {

        int mx = INT_MIN, mn = INT_MAX;

        // 记录出现次数，用于判断去重(值域是可以计算出来的)

        int freq[200000] = {0};

        int dupFlag = 0;

 

        for (int j = i - m; j < i; j++) {

            if (freq[num[j]] == 1) {

                dupFlag = 1;

            }

            freq[num[j]]++;

 

            if (num[j] > mx) mx = num[j];

            if (num[j] < mn) mn = num[j];

        }

 

        if (dupFlag) {

            num[i] = mx + mn;

        } else {

            num[i] = mx - mn;

        }

    }

 

    return num[n];

}

 

int main() {

    int m, n;

    scanf("%d,%d", &m, &n);

 

    printf("%d\n", positionValue(m, n));

    return 0;

}
