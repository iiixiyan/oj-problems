100分-文章特征提取

题目描述

一篇文档由多个文本片段组成。这里假定每个片段只由小写英文字母组成，不包含其他内容。现在需要对这些文本片段进行特征提取，规则如下：

  1. 先找出所有片段中都包含的字母，也就是该字母在每个片段中都至少出现过一次。
  2. 如果某个字母在多个片段中都出现过，则取它在所有片段中出现次数的最小值。
    • 若最小值为 1，则结果中包含 1 个该字母
    • 若最小值为 2，则结果中包含 2 个该字母
    • 以此类推
  3. 最终输出满足条件的所有字母组成的字符串，并按字母从小到大排序。
  4. 如果没有任何字母满足条件，则结果为空字符串。

补充说明

  • 输入中的字符串（即片段）总数小于 1000

输入描述

输入为若干文本片段，片段之间用空格分割。

输出描述

输出特征提取结果。

用例1

输入

paper parent parade

输出

aepr

说明
p、a、e、r 在每个字符串中都至少出现一次，且每个字母在各字符串中的最小出现次数都是 1，所以结果为 aepr。

用例2

输入

hello hollow halloween

输出

hllo

说明

  • h 在三个字符串中的出现次数最小值为 1
  • l 在三个字符串中的出现次数最小值为 2
  • o 在三个字符串中的出现次数最小值为 1

所以结果为 hllo。

用例3

输入

abc def ghi

输出

说明
没有字母在所有三个字符串中都出现。

题解

思路：模拟 / 计数

  1. 统计每个文本片段中 26 个字母的出现次数。
  2. 对每个字母，取它在所有片段中的出现次数最小值。
  3. 如果最小值大于 0，就把该字母加入答案对应次数。
  4. 按字母从小到大拼接输出结果。

// 通用 切割函数 函数 将字符串str根据delimiter进行切割

vector<string> split(const string& str, const string& delimiter) {

    vector<string> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(str.substr(start, end - start));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(str.substr(start));

    return result;

}

 

string featureExtraction(vector<string>& docs) {

    vector<int> minCount(26, INT_MAX);

    for (int i = 0; i < docs.size(); i++) {

        vector<int> curCount(26, 0);

        // 统计出现次数

        for (auto& c : docs[i]) {

            curCount[c - 'a']++;

        }

 

        // 更新全局最小值

        for (int j = 0; j < 26; j++) {

            minCount[j] = min(minCount[j], curCount[j]);

        }

    }

   // 构建结果

    string res;

    for (int i = 0; i < 26; i++) {

        res.append(minCount[i], char('a' + i));

    }

    return res;

}

 

 

int main() {

    string input;

    getline(cin, input);

    vector<string> docs = split(input, " ");

    string res = featureExtraction(docs);

    cout << res;

    return 0;

}

// 功能：提取所有字符串的公共字符（最小出现次数）

function featureExtraction(docs) {

    let minCount = new Array(26).fill(Infinity);

 

    for (let doc of docs) {

        let curCount = new Array(26).fill(0);

 

        // 统计当前字符串字符出现次数

        for (let c of doc) {

            curCount[c.charCodeAt(0) - 'a'.charCodeAt(0)]++;

        }

 

        // 更新全局最小值

        for (let i = 0; i < 26; i++) {

            minCount[i] = Math.min(minCount[i], curCount[i]);

        }

    }

 

    // 构建结果

    let res = '';

    for (let i = 0; i < 26; i++) {

        res += String.fromCharCode('a'.charCodeAt(0) + i).repeat(minCount[i]);

    }

 

    return res;

}

#include <stdio.h>

#include <string.h>

#include <limits.h>

 

// 功能：提取所有字符串的公共字符（最小出现次数）

void featureExtraction(char docs[][1000], int n, char* res) {

    int minCount[26];

    for (int i = 0; i < 26; i++) {

        minCount[i] = INT_MAX;

    }

 

    for (int i = 0; i < n; i++) {

        int curCount[26] = {0};

 

        // 统计当前字符串字符出现次数

        for (int j = 0; docs[i][j] != '\0'; j++) {

            curCount[docs[i][j] - 'a']++;

        }

 

        // 更新全局最小值

        for (int j = 0; j < 26; j++) {

            if (curCount[j] < minCount[j]) {

                minCount[j] = curCount[j];

            }

        }

    }

 

    // 构建结果

    int idx = 0;

    for (int i = 0; i < 26; i++) {

        for (int j = 0; j < minCount[i]; j++) {

            res[idx++] = 'a' + i;

        }

    }

    res[idx] = '\0';

}

 

int main() {

    char input[10000000];

    fgets(input, sizeof(input), stdin);

 

    // 去掉换行

    input[strcspn(input, "\n")] = 0;

 

    char docs[1010][1000];

    int count = 0;

 

    char* token = strtok(input, " ");

    while (token != NULL) {

        strcpy(docs[count++], token);

        token = strtok(NULL, " ");

    }

 

    char res[100];

    featureExtraction(docs, count, res);

 

    printf("%s", res);

    return 0;

}
