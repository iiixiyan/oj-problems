100分-配置操作失败数量统计 (Py/Java/C/C++/Js/Go)

题目描述

模拟一个系统的命令行配置，包含添加、修改、删除三项操作，详情如下：

  • 添加操作命令：add_rule rule_id=1 rule_index=18
  • 修改操作命令：mod_rule rule_id=1 rule_index=100
  • 删除操作命令：del_rule rule_id=1

其中：

  • add_rule、mod_rule、del_rule 是操作关键字
  • rule_id 和 rule_index 是属性关键字
  • 属性取值范围为数字 1 ~ 9999
  • 操作、属性之间都用空格进行分割

规则说明

  1. 在进行所有操作时，如果缺少关键字，或者相应的 rule_id、rule_index 的取值不符合要求，则操作失败。
  2. 在进行添加操作时，参数必须包含 rule_id 和 rule_index。如果当前不存在该 rule_id，则添加成功；如果添加已经存在的 rule_id，则操作失败。
  3. 在进行修改操作时，参数必须包含 rule_id 和 rule_index。如果当前 rule_id 不存在，或前后 rule_index 没有变化，则操作失败。
  4. 在进行删除操作时，参数必须包含 rule_id。如果当前 rule_id 不存在，则操作失败。

批量操作说明

在进行批量操作时，一个命令失败后仍可以继续下一条命令的操作。
现给有一组批量操作的字符串，包括不超过 1000 条连续的操作指令，格式为：

[cmd][cmd][cmd]

请将字符串解析后按照顺序进入你实现的系统，统计出配置失败的次数。

输入描述

输入命令以空格分割。

输出描述

输出失败命令数量。

用例1

输入

add_rule rule_id=1 rule_index=9999

mod_rule rule_id=1 rule_index=10

del_rule rule_id=1

输出

0

说明
所有操作都成功。

用例2

输入

add_rule rule_id=1

mod_rule rule_id=1 rule_index=10

del_rule rule_id=1

输出

3

说明
add 操作不包含 rule_index，添加失败；
后续修改和删除操作，由于无对应 rule_id 数据，也会失败。

用例3

输入

add_rule rule_id=1 rule_index=10000

输出

1

说明
rule_index 超过范围。

题解

思路：模拟

  1. 使用哈希表或者类似结构存储当前已创建的 rule_id 对应的 rule_index。
  2. 循环遍历输入命令字符串，依次判断：
    • 操作关键字是否正确
    • 操作关键字 / 属性关键字是否缺失
    • rule_id、rule_index 是否在合法范围内
    • 添加 / 删除时，rule_id 是否已存在或不存在
    • 修改时，rule_id 是否存在，且修改后的值是否与原值不同
  3. 按照上述逻辑处理每个命令，累计失败次数并输出。

// 通用 切割函数 函数 将字符串str根据delimiter进行切割

vector<string> split(const string& str, const string& delimiter) {

    vector<string> result;

    size_t start = 0;

    size_t end = str.find(delimiter);

    while (end != string::npos) {

        result.push_back(str.substr(start, end - start));

        start = end + delimiter.length();

        end = str.find(delimiter, start);

    }

    // 添加最后一个部分

    result.push_back(str.substr(start));

    return result;

}

 

bool judegeNumber(string& s) {

    // 1. 不能为空

    if (s.empty()) return false;

 

    // 2. 判断是否全是数字

    for (char c : s) {

        if (!isdigit(c)) return false;

    }

 

    // 3. 转成整数

    int num = stoi(s);

 

    // 4. 判断范围

    return num >= 1 && num <= 9999;

}

public class Main {

 

    // 判断字符串是否为 1~9999 的整数

    static boolean judegeNumber(String s) {

        if (s.isEmpty()) return false;

        for (char c : s.toCharArray()) {

            if (!Character.isDigit(c)) return false;

        }

        int num = Integer.parseInt(s);

        return num >= 1 && num <= 9999;

    }

 

    // 核心逻辑函数：计算错误命令数量

    static int calErrorCmdError(List<String> command) {

        int errorCount = 0;

        Map<String, String> ruleIndex = new HashMap<>(); // 存储 rule_id -> rule_index

 

        for (String line : command) {

            String[] cmd = line.split(" "); // 按空格拆分命令

 

            // 命令长度异常

            if (cmd.length != 2 && cmd.length != 3) {

                errorCount++;

                continue;

            }

 

            String operatorCmd = cmd[0];

 

            if (operatorCmd.equals("del_rule")) { // 删除规则

                if (cmd.length != 2) { errorCount++; continue; }

 

                String[] idPropertys = cmd[1].split("=");

                if (idPropertys.length != 2 || !idPropertys[0].equals("rule_id")) {

                    errorCount++; continue;

                }

 

                String id = idPropertys[1];

                // 检查 id 是否有效且存在

                if (!judegeNumber(id) || !ruleIndex.containsKey(id)) {

                    errorCount++; continue;

                }

 

                // 正常删除

                ruleIndex.remove(id);

 

            } else if (operatorCmd.equals("mod_rule") || operatorCmd.equals("add_rule")) { // 修改或添加规则

                if (cmd.length != 3) { errorCount++; continue; }

 

                String[] idPropertys = cmd[1].split("=");

                String[] indexPropertys = cmd[2].split("=");

 

                // 检查格式

                if (idPropertys.length != 2 || indexPropertys.length != 2

                        || !idPropertys[0].equals("rule_id") || !indexPropertys[0].equals("rule_index")) {

                    errorCount++; continue;

                }

 

                String id = idPropertys[1];

                String index = indexPropertys[1];

 

                // 检查数字范围

                if (!judegeNumber(id) || !judegeNumber(index)) { errorCount++; continue; }

 

                if (operatorCmd.equals("mod_rule")) {

                    // 修改规则：不存在或值没变算错误

                    if (!ruleIndex.containsKey(id) || ruleIndex.get(id).equals(index)) { errorCount++; continue; }

                    ruleIndex.put(id, index);

                } else {

                    // 添加规则：已存在算错误

                    if (ruleIndex.containsKey(id)) { errorCount++; continue; }

                    ruleIndex.put(id, index);

                }

 

            } else {

                // 操作关键字不对

                errorCount++;

            }

        }

 

        return errorCount;

    }

 

    public static void main(String[] args) {

// 判断字符串是否为 1~9999 的整数

function judegeNumber(s) {

    if (!/^\d+$/.test(s)) return false;

    let num = parseInt(s);

    return num >= 1 && num <= 9999;

}

 

// 核心逻辑函数：计算错误命令数量

function calErrorCmdError(commands) {

    let errorCount = 0;

    let ruleIndex = {}; // 存储 rule_id -> rule_index

 

    for (let line of commands) {

        let cmd = line.trim().split(/\s+/);

        if (cmd.length !== 2 && cmd.length !== 3) { errorCount++; continue; }

 

        let operatorCmd = cmd[0];

 

        if (operatorCmd === "del_rule") { // 删除规则

            if (cmd.length !== 2) { errorCount++; continue; }

            let idPropertys = cmd[1].split("=");

            if (idPropertys.length !== 2 || idPropertys[0] !== "rule_id") { errorCount++; continue; }

            let id = idPropertys[1];

            if (!judegeNumber(id) || !(id in ruleIndex)) { errorCount++; continue; }

            delete ruleIndex[id];

 

        } else if (operatorCmd === "mod_rule" || operatorCmd === "add_rule") { // 修改或添加规则

            if (cmd.length !== 3) { errorCount++; continue; }

            let idPropertys = cmd[1].split("=");

            let indexPropertys = cmd[2].split("=");

            if (idPropertys.length !== 2 || indexPropertys.length !== 2

                || idPropertys[0] !== "rule_id" || indexPropertys[0] !== "rule_index") { errorCount++; continue; }

 

            let id = idPropertys[1];

            let index = indexPropertys[1];

            if (!judegeNumber(id) || !judegeNumber(index)) { errorCount++; continue; }

 

            if (operatorCmd === "mod_rule") {

                if (!(id in ruleIndex) || ruleIndex[id] === index) { errorCount++; continue; }

                ruleIndex[id] = index;

            } else { // add_rule

                if (id in ruleIndex) { errorCount++; continue; }

                ruleIndex[id] = index;

            }

 

        } else {

            // 操作关键字不对

            errorCount++;

        }

    }

    return errorCount;

}

package main

 

import (

    "bufio"

    "fmt"

    "os"

    "strconv"

    "strings"

)

 

// 判断字符串是否为 1~9999 的整数

func judegeNumber(s string) bool {

    if s == "" { return false }

    for _, c := range s {

        if c < '0' || c > '9' { return false }

    }

    num, _ := strconv.Atoi(s)

    return num >= 1 && num <= 9999

}

 

// 计算错误命令数量

func calErrorCmdError(commands []string) int {

    errorCount := 0

    ruleIndex := make(map[string]string) // 存储 rule_id -> rule_index

 

    for _, line := range commands {

        cmd := strings.Fields(line)

        if len(cmd) != 2 && len(cmd) != 3 { errorCount++; continue }

        operatorCmd := cmd[0]

 

        if operatorCmd == "del_rule" { // 删除规则

            if len(cmd) != 2 { errorCount++; continue }

            kv := strings.Split(cmd[1], "=")

            if len(kv) != 2 || kv[0] != "rule_id" { errorCount++; continue }

            id := kv[1]

            if !judegeNumber(id) || ruleIndex[id] == "" { errorCount++; continue }

            delete(ruleIndex, id)

        } else if operatorCmd == "mod_rule" || operatorCmd == "add_rule" { // 修改或添加规则

            if len(cmd) != 3 { errorCount++; continue }

            idKV := strings.Split(cmd[1], "=")

            indexKV := strings.Split(cmd[2], "=")

            if len(idKV) != 2 || len(indexKV) != 2 || idKV[0] != "rule_id" || indexKV[0] != "rule_index" {

                errorCount++; continue

            }

            id := idKV[1]

            index := indexKV[1]

            if !judegeNumber(id) || !judegeNumber(index) { errorCount++; continue }

 

            if operatorCmd == "mod_rule" {

                if ruleIndex[id] == "" || ruleIndex[id] == index { errorCount++; continue }

                ruleIndex[id] = index

            } else {

                if ruleIndex[id] != "" { errorCount++; continue }

                ruleIndex[id] = index

            }

        } else {

            // 操作关键字不对

            errorCount++

        }

    }

 

    return errorCount

}

#include <stdio.h>

#include <string.h>

#include <stdlib.h>

 

#define MAXN 1010

#define MAXLEN 50

 

// 判断字符串是否为 1~9999 的整数

int judegeNumber(char *s) {

    if (*s == '\0') return 0;

    for (int i = 0; s[i]; i++) {

        if (s[i] < '0' || s[i] > '9') return 0;

    }

    int num = atoi(s);

    return num >= 1 && num <= 9999;

}

 

// 查找 id 在数组中的下标

int find(char ids[][MAXLEN], int size, char *id) {

    for (int i = 0; i < size; i++) {

        if (strcmp(ids[i], id) == 0) return i;

    }

    return -1;

}

 

// 核心逻辑函数：计算错误命令数量

int calErrorCmdError(int n, char cmds[][3*MAXLEN]) {

    int errorCount = 0;

    char ids[MAXN][MAXLEN];     // 存储 rule_id

    char indexes[MAXN][MAXLEN]; // 存储 rule_index

    int size = 0;               // 当前已有规则数量

 

    for (int i = 0; i < n; i++) {

        char *cmd = cmds[i];

 

        // 拆分命令

        char parts[3][MAXLEN];

        int partCount = 0;

        char *p = strtok(cmd, " ");

        while (p != NULL && partCount < 3) {

            strcpy(parts[partCount++], p);

            p = strtok(NULL, " ");

        }

 

        if (partCount != 2 && partCount != 3) { errorCount++; continue; }

 

        if (strcmp(parts[0], "del_rule") == 0) { // 删除规则

            if (partCount != 2) { errorCount++; continue; }

            char *k = strtok(parts[1], "=");

            char *v = strtok(NULL, "=");

            if (!k || !v || strcmp(k,"rule_id") != 0 || !judegeNumber(v)) { errorCount++; continue; }

            int idx = find(ids, size, v);

            if (idx == -1) { errorCount++; continue; }

            // 删除：用最后一个元素覆盖

            if (idx != size-1) {

                strcpy(ids[idx], ids[size-1]);

                strcpy(indexes[idx], indexes[size-1]);

            }

            size--;

        } else if (strcmp(parts[0], "mod_rule") == 0 || strcmp(parts[0], "add_rule") == 0) { // 修改或添加

            if (partCount != 3) { errorCount++; continue; }

            char *idKV = parts[1];

            char *idxKV = parts[2];

            char *k1 = strtok(idKV, "="); char *v1 = strtok(NULL, "=");

            char *k2 = strtok(idxKV, "="); char *v2 = strtok(NULL, "=");

            if (!k1 || !v1 || !k2 || !v2 || strcmp(k1,"rule_id")!=0 || strcmp(k2,"rule_index")!=0

                || !judegeNumber(v1) || !judegeNumber(v2)) { errorCount++; continue; }

 

            int idx = find(ids, size, v1);

            if (strcmp(parts[0],"mod_rule")==0) { // 修改

                if (idx == -1 || strcmp(indexes[idx], v2)==0) { errorCount++; continue; }

                strcpy(indexes[idx], v2);

            } else { // 添加

                if (idx != -1) { errorCount++; continue; }

                strcpy(ids[size], v1);

                strcpy(indexes[size], v2);

                size++;

            }

 

        } else { // 操作关键字错误

            errorCount++;

        }

    }

 

    return errorCount;

}

 

int main() {

    char line[3*MAXLEN];

    char cmds[MAXN][3*MAXLEN];

    int n = 0;

    while (fgets(line, sizeof(line), stdin)) {

        if (line[0]=='\n' || line[0]=='\0') break;

        line[strcspn(line,"\n")] = 0; // 去掉换行符

        strcpy(cmds[n++], line);

    }

 

    int res = calErrorCmdError(n, cmds);

    printf("%d\n", res);

    return 0;

}
